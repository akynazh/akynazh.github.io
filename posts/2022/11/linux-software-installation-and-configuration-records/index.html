安装

 1yum update
 2yum install openssl-devel bzip2-devel libffi-devel -y

 4wget https://www.python.org/ftp/python/3.10.2/Python-3.10.2.tgz

 8make altinstall

lzma

ModuleNotFoundError: No module named ‘_lzma’ 解決方法：

 1yum install xz-devel
 2yum install python-backports-lzma
 3pip install backports.lzma
 4
 5找到lzma.py，如日誌提示的/usr/local/lib/python3.7/lzma.py，修改zma.py文件中的導入部份如下：
 6
 7修改前：
 8from _lzma import *
 9from _lzma import _encode_filter_properties, _decode_filter_properties
10
11修改後：
12try:
13    from _lzma import *
14    from _lzma import _encode_filter_properties, _decode_filter_properties
15except ImportError:
16    from backports.lzma import *
17    from backports.lzma import _encode_filter_properties, _decode_filter_properties

2tar -zxvf jdk-8u151-linux-x64.tar.gz
3sudo mkdir /usr/local/java
4sudo mv jdk1.8.0_151 /usr/local/java/jdk1.8

环境变量

1# java env
2export JAVA_HOME=/usr/local/java/jdk1.8
3export CLASSPATH=.:$JAVA_HOME/lib:$JAVA_HOME/jre/lib
4export PATH=$PATH:$JAVA_HOME/bin
