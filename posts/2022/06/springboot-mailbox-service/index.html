SpringBoot 邮箱服务

2022-06-16
3 minute read
Web
springboot

一、导入依赖：

1<!--发送邮件-->
2<dependency>
3    <groupId>org.springframework.boot</groupId>
4    <artifactId>spring-boot-starter-mail</artifactId>
5</dependency>

二、配置文件：

QQ邮箱需要先开通好邮箱SMTP服务。

 1spring:
 2  # 邮件服务
 3  mail:
 4    password: xxxxxx
 5    username: xxx@qq.com
 6    host: smtp.QQ.com
 7    properties:
 8      mail:
 9        smtp:
10          auth: true
11          starttls:
12            enable: true
13            required: true
14    default-encoding: UTF-8

三、编写邮件服务

  • sendSimpleMailMessage()实现只含普通文本信息的邮件发送
  • sendMimeMessage()可以发送带附件的邮件

MailService.java

 1@Service
 2public class MailService {
 3    private final JavaMailSender mailSender;
 4    @Value("${spring.mail.username}")
 5    private String SENDER;
 6
 7    public MailService(JavaMailSender mailSender) {
 8        this.mailSender = mailSender;
 9    }
10
11    // 发送普通邮件
12    public void sendSimpleMailMessage(String to, String subject, String content) {
13        SimpleMailMessage message = new SimpleMailMessage();
14        message.setFrom(SENDER);
15        message.setTo(to);
16        message.setSubject(subject);
17        message.setText(content);
18        mailSender.send(message);
19    }
20
21    // 发送带附件的邮件
22    public void sendMimeMessage(String to, String subject, String content, String filePath) throws MessagingException {
23        MimeMessage message = mailSender.createMimeMessage();
24        // true表示需要创建一个multipart message
25        MimeMessageHelper helper = new MimeMessageHelper(message, true);
26        helper.setFrom(SENDER);
27        helper.setTo(to);
28        helper.setSubject(subject);
29        helper.setText(content, true);
30
31        FileSystemResource file = new FileSystemResource(new File(filePath));
32        String fileName = file.getFilename();
33        helper.addAttachment(fileName, file);
34        mailSender.send(message);
35    }
36}

四、编写验证码生成器

传入参数为验证码长度。

 1public class VerifyCodeGenerator {
 2    public static String make(int length) {
 3        if (length <= 0) {
 4            throw new RuntimeException("验证码长度不能小于1");
 5        }
 6        StringBuilder sb = new StringBuilder();
 7        Random random = new Random();
 8        for (int i = 0; i < length; i++) {
 9            sb.append(random.nextInt(10));
10        }
11        return sb.toString();
12    }
13}

五、Controller编写

接受用户输入的邮箱，并向该邮箱发送一个验证码；

检验用户输入的验证码，如果正确则放行。

 1@RestController
 2@RequestMapping("/verify")
 3public class VerifyTest {
 4    private MailService mailService;
 5    @Autowired
 6    public void setMailService(MailService mailService) {
 7        this.mailService = mailService;
 8    }
 9
10    @PostMapping("/login/{email}/{code}")
11    public String login(@PathVariable String email, @PathVariable String code, HttpSession session) {
12        if (StringUtils.isEmpty(email) || StringUtils.isEmpty(code)) {
13            return "false";
14        }
15        String trueCode = (String)session.getAttribute("code");
16        if (code.equals(trueCode)) {
17            return "success";
18        } else {
19            return "fail";
20        }
21    }
22    @PostMapping("/{email}")
23    public String verify(@PathVariable String email, HttpSession session) {
24        if (StringUtils.isEmpty(email)) {
25            return "false";
26        }
27        String code = VerifyCodeGenerator.make(6);
28        session.setAttribute("code", code);
29        mailService.sendSimpleMailMessage(email, "Email Verify", "Your code: " + code);
30        return email;
31    }
32}
